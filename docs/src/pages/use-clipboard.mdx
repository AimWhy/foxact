# useClipboard

import ExportMetaInfo from '../components/export-meta-info';

<ExportMetaInfo />

Handle copying content to clipboard with user-friendly feedbacks.

## Usage

```tsx
import { useClipboard } from 'foxact/use-clipboard';

const Demo = () => {
  const { copy, copied, error } = useClipboard({
    // optional, default to 1000, the duration of the copied state after copying
    timeout: 1000,
    // optional, default to false, whether to use `window.prompt` as fallback when copying failed
    usePromptAsFallback: false
    // optional. When `window.prompt` is used as fallback, this text will be shown in the prompt dialog
    promptFallbackText: 'Failed to copy to clipboard automatically, please manually copy the text below.'
  });

  return (
    <button onClick={useCallback(() => copy('sukka'), [copy])}>
      {copied ? 'Copied' : 'Copy'}
    </button>
  );
}
```
